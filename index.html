  <script>

    const startBtn = document.getElementById('start-btn');

    const startScreen = document.getElementById('start-screen');

    const arContainer = document.getElementById('ar-container');

    let audioCtx = null;


    startBtn.addEventListener('click', async () => {

      // 1. Handle Audio Context for Mobile

      audioCtx = new (window.AudioContext || window.webkitAudioContext)();

      if (audioCtx.state === 'suspended') await audioCtx.resume();

      

      // 2. Play a tiny silent beep to 'unlock' audio on iOS

      const osc = audioCtx.createOscillator();

      const gain = audioCtx.createGain();

      gain.gain.value = 0.01;

      osc.connect(gain); gain.connect(audioCtx.destination);

      osc.start(); osc.stop(audioCtx.currentTime + 0.01);


      // 3. Hide Splash and Show AR

      startScreen.style.display = 'none';

      arContainer.style.display = 'block';


      // 4. Trigger A-Frame to resize (helps with the camera alignment)

      window.dispatchEvent(new Event('resize'));

      

      console.log("AR Scene Initialized");

      setupSceneContent();

    });


    function setupSceneContent() {

      const container = document.querySelector('#marker-content');

      

      // Simple Sphere Test for Hiro Marker

      const sphere = document.createElement('a-sphere');

      sphere.setAttribute('position', '0 0.5 0');

      sphere.setAttribute('radius', '0.5');

      sphere.setAttribute('color', '#EF2D5E');

      sphere.setAttribute('animation', 'property: position; to: 0 1.5 0; dir: alternate; dur: 2000; loop: true');

      

      container.appendChild(sphere);

    }

  </script>
