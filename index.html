  <script>

    // --- AUDIO & TEXTURE LOGIC FROM ORIGINAL FILE ---

    let audioCtx, shapeSound, masterOutput;

    const startBtn = document.getElementById('start-btn');


    function initAudio() {

      audioCtx = new (window.AudioContext || window.webkitAudioContext)();

      const limiter = audioCtx.createDynamicsCompressor();

      limiter.threshold.value = -1; 

      limiter.connect(audioCtx.destination);

      const droneGain = audioCtx.createGain();

      droneGain.gain.value = 0.20;

      [[55,'sine'],[55.5,'sine'],[82.4,'triangle']].forEach(([f, t]) => {

        const o = audioCtx.createOscillator(); o.type = t; o.frequency.value = f;

        o.connect(droneGain); o.start();

      });

      droneGain.connect(limiter);

      return limiter;

    }


    function makeLinearTexture(bg, fg, words, fontSize) {

      const size = 1024;

      const c = document.createElement('canvas'); c.width = size; c.height = size;

      const ctx = c.getContext('2d');

      ctx.fillStyle = bg; ctx.fillRect(0, 0, size, size);

      ctx.font = `900 ${fontSize}px Arial Black`; ctx.fillStyle = fg;

      words.forEach((w, i) => ctx.fillText(w, 50, 200 + i * 200));

      return new THREE.CanvasTexture(c);

    }


    // --- INITIALIZATION ---

    startBtn.addEventListener('click', async () => {

      masterOutput = initAudio();

      if (audioCtx.state === 'suspended') await audioCtx.resume();

      

      document.getElementById('start-screen').style.display = 'none';

      document.getElementById('ar-container').style.display = 'block';

      window.dispatchEvent(new Event('resize'));

      setupShapes();

    });


    let shapes = [], trajectories = [], startTime, lastIdx = -1;


    function setupShapes() {

      const group = document.getElementById('ar-objects').object3D;

      const mat = (t) => new THREE.MeshStandardMaterial({ map: t });

      

      const s1 = new THREE.Mesh(new THREE.SphereGeometry(1, 32, 32), mat(makeLinearTexture('#fff','#000',['SPACE','TIME'],120)));

      const s2 = new THREE.Mesh(new THREE.BoxGeometry(1.5,1.5,1.5), mat(makeLinearTexture('#ff69b4','#000',['IDEA'],120)));

      const s3 = new THREE.Mesh(new THREE.TorusGeometry(1, 0.4), mat(makeLinearTexture('#ff6600','#000',['GO'],120)));


      shapes = [s1, s2, s3];

      trajectories = [{x:8, z:-20}, {x:-8, z:-20}, {x:0, z:-10}]; // Simplied paths: Right, Left, Center

      

      shapes.forEach(s => { s.visible = false; group.add(s); });

      group.add(new THREE.AmbientLight(0xffffff, 1.2));

      

      animate();

    }


    function animate() {

      requestAnimationFrame(animate);

      if (!startTime) startTime = performance.now();

      const elapsed = (performance.now() - startTime) % 30000;

      const idx = Math.floor(elapsed / 10000);

      const t = (elapsed % 10000) / 10000;


      shapes.forEach((s, i) => s.visible = (i === idx));

      const s = shapes[idx];

      const traj = trajectories[idx];

      

      // Animation Logic

      s.position.z = -25 + (t * 30);

      s.position.x = (t > 0.3) ? (t-0.3) * traj.x : 0;

      s.rotation.y += 0.02;

      s.scale.setScalar(0.5 + t * 4);

    }
