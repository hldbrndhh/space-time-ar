  <script>

    let audioCtx = null, shapeSound = null, masterOutput = null;


    function initAudio() {

      audioCtx = new (window.AudioContext || window.webkitAudioContext)();

      const limiter = audioCtx.createDynamicsCompressor();

      limiter.threshold.value = -1; 

      limiter.connect(audioCtx.destination);

      const droneGain = audioCtx.createGain();

      droneGain.gain.value = 0.0;

      [[55,'sine'],[55.5,'sine'],[82.4,'triangle']].forEach(([freq, type]) => {

        const o = audioCtx.createOscillator(); 

        o.type = type; o.frequency.value = freq;

        o.connect(droneGain); o.start();

      });

      droneGain.connect(limiter);

      droneGain.gain.setTargetAtTime(0.20, audioCtx.currentTime, 2.5);

      return limiter;

    }


    const shapeSoundConfigs = [

      { freq: 220, type: 'sine', filterFreq: 800 }, { freq: 174.6, type: 'square', filterFreq: 600 },

      { freq: 261.6, type: 'triangle', filterFreq: 1200 }, { freq: 196.0, type: 'sine', filterFreq: 900 },

      { freq: 293.7, type: 'triangle', filterFreq: 1400 }, { freq: 130.8, type: 'square', filterFreq: 500 }

    ];


    function startShapeSound(index) {

      if (!audioCtx) return;

      stopShapeSound();

      const cfg = shapeSoundConfigs[index];

      const gain = audioCtx.createGain(); gain.gain.value = 0.0;

      const osc = audioCtx.createOscillator(); osc.type = cfg.type; osc.frequency.value = cfg.freq;

      const filter = audioCtx.createBiquadFilter(); filter.type = 'lowpass'; filter.frequency.value = cfg.filterFreq;

      osc.connect(filter); filter.connect(gain); gain.connect(masterOutput);

      osc.start(); shapeSound = { osc, gain };

    }


    function setShapeSoundVolume(t) {

      if (!shapeSound) return;

      let vol = (t < 0.6) ? (t / 0.6) * 0.45 : (t < 0.8) ? 0.45 : ((1 - t) / 0.2) * 0.45;

      shapeSound.gain.gain.setTargetAtTime(vol, audioCtx.currentTime, 0.1);

    }


    function stopShapeSound() {

      if (shapeSound) { try { shapeSound.osc.stop(); } catch(e) {} shapeSound = null; }

    }


    function makeLinearTexture(bg, fg, words, fontSize) {

      const size = 1024;

      const c = document.createElement('canvas'); c.width = size; c.height = size;

      const ctx = c.getContext('2d');

      ctx.fillStyle = bg; ctx.fillRect(0, 0, size, size);

      ctx.font = `bold ${fontSize}px Arial`; ctx.fillStyle = fg;

      words.forEach((w, i) => ctx.fillText(w, 50, 150 + i * 150));

      return new THREE.CanvasTexture(c);

    }


    let arScene, markerRoot, shapes = [], startTime = null, lastShapeIndex = -1;


    async function startAR() {

      document.getElementById('start-screen').style.display = 'none';

      document.getElementById('info').style.display = 'block';

      masterOutput = initAudio();

      if (audioCtx.state === 'suspended') await audioCtx.resume();

      initARScene();

    }


    function initARScene() {

      const sceneEl = document.createElement('a-scene');

      sceneEl.setAttribute('embedded', '');

      sceneEl.setAttribute('arjs', 'sourceType: webcam; debugUIEnabled: false;');

      document.body.appendChild(sceneEl);


      sceneEl.addEventListener('loaded', () => {

        const markerEl = document.createElement('a-marker');

        // --- CHANGE IS HERE ---

        markerEl.setAttribute('preset', 'hiro'); 

        // ----------------------

        sceneEl.appendChild(markerEl);


        markerRoot = new THREE.Group();

        markerEl.object3D.add(markerRoot);

        markerRoot.add(new THREE.AmbientLight(0xffffff, 1.0));


        const mat = (tex) => new THREE.MeshStandardMaterial({ map: tex });

        shapes = [

          new THREE.Mesh(new THREE.SphereGeometry(1), mat(makeLinearTexture('#fff','#000',['SPACE','TIME'],100))),

          new THREE.Mesh(new THREE.BoxGeometry(1.5,1.5,1.5), mat(makeLinearTexture('#f0f','#000',['IDEA'],100))),

          new THREE.Mesh(new THREE.TorusGeometry(1, 0.4), mat(makeLinearTexture('#0ff','#000',['GO'],100))),

          new THREE.Mesh(new THREE.IcosahedronGeometry(1.2), mat(makeLinearTexture('#ff0','#000',['SMILE'],100)))

        ];

        shapes.forEach(s => { s.visible = false; markerRoot.add(s); });

        animate();

      });

      const cam = document.createElement('a-entity'); cam.setAttribute('camera',''); sceneEl.appendChild(cam);

    }


    function animate() {

      requestAnimationFrame(animate);

      const elapsed = (performance.now() - (startTime || (startTime = performance.now()))) % 40000;

      const idx = Math.floor(elapsed / 10000) % shapes.length;

      const t = (elapsed % 10000) / 10000;


      if (idx !== lastShapeIndex) { lastShapeIndex = idx; startShapeSound(idx); }

      shapes.forEach((s, i) => s.visible = (i === idx));

      shapes[idx].position.z = -15 + (t * 20);

      shapes[idx].rotation.y += 0.01;

      setShapeSoundVolume(t);

    }
